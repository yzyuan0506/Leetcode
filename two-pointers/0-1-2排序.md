一个数组只有0，1，2三种元素，不允许使用额外空间，每个元素只许遍历一次，将这个数组从小到大排序。

```cpp
void sort(int arr[],int len)
{
    int i = 0;  // 头指针指向下一个 0 应该在的位置
    int  j = len - 1;//尾指针指向下一个 2 应该在的位置
    int k = 0;

    while (arr[i] == 0)
        i++;
    k = i + 1;

    while (arr[j] == 2)
        j--;

    while (k < j)
    {
        if (arr[k] == 1)
            k++;
        else if (arr[k] == 0)
        {
            swap(arr[i], arr[k]);
            while (arr[i] == 0)
                i++;
        }
        else if (arr[k] == 2)
        {
            swap(arr[k], arr[j]);
            while (arr[j] == 2)
                j--;
        }
    }
}
```
