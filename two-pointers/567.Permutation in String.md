```
class Solution {
public:
    bool checkInclusion(string s1, string s2) {
        int l1 = s1.size();
        int l2 = s2.size();
        if (l1 > l2) return false;
        
        vector<int> c1(26, 0);
        vector<int> c2(26, 0);

        for (int i = 0; i < l1; i++) {
            c1[s1[i] - 'a'] += 1;
        }
        

        for (int i = 0, n = 0; i < l2; i++) {
            
            c2[s2[i] - 'a'] += 1;
            n += 1;
            if (n > l1) {
                n -= 1;
                c2[s2[i - n] - 'a'] -= 1;   
            }
            if (c1 == c2) return true;  // 时间复杂度 26 - O(1)
            
        }
        return false;
    }
};
```

听说这方法叫 sliding window
将 字母 分别放在两个bucket里，比较bucket是否一样就行。。

时间复杂度 O(l1 + l2)
空间复杂度 O(1)


甚至简化成一个pointer。。
```
class Solution {
public:
    bool checkInclusion(string s1, string s2) {
        int l1 = s1.size();
        int l2 = s2.size();
        if (l1 > l2) return false;
        
        vector<int> c1(26, 0);
        vector<int> c2(26, 0);

        for (int i = 0; i < l1; i++) {
            c1[s1[i] - 'a'] += 1;
        }
        

        for (int i = 0; i < l2; i++) {
            
            if (i >= l1) {
                c2[s2[i - l1] - 'a'] -= 1;
            }
            c2[s2[i] - 'a'] += 1;

            if (c1 == c2) return true;  // 时间复杂度 26 - O(1)
            
        }
        return false;
    }
};
```
