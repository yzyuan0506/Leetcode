```
public void moveZeroes(int[] nums) {
    if (nums == null || nums.length == 0) return;        

    // 在能插入的位置(不是零的位置)插入数字
    int insertPos = 0;
    for (int num: nums) {
        if (num != 0) nums[insertPos++] = num;
    }        

    while (insertPos < nums.length) {
        nums[insertPos++] = 0;
    }
}
```

// 更好的方法，避免swap相同的数字
// 填坑之后，把原来的位置换为0， 因为 pos <= i
```
class Solution {
    public void moveZeroes(int[] nums) {
        int pos = 0;
        for(int i = 0;i < nums.length;i++) {
            if(nums[i] != 0) {
                if(i != pos) {
                    nums[pos] = nums[i];
                    nums[i] = 0;
                }
                pos++;
            }
        }
    }
}
```
