class Solution {
public:
    TreeNode* invertTree(TreeNode* root) {
        if(!root) return NULL;

        queue<TreeNode*> q;
        q.push(root);
        while(!q.empty()) {

            for(int i = 0; i < q.size(); i++) {
                TreeNode* node = q.front();
                q.pop();
                TreeNode* temp = node->left;
                node->left = node->right;
                node->right = temp;
                if(node->left) q.push(node->left);
                if(node->right) q.push(node->right);
            }
        }
        return root;
    }
};
