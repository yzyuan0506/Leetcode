// 方法1：递归
// Preorder traversal: (root, left, right)
```
class Solution {
public:
    TreeNode* mergeTrees(TreeNode* t1, TreeNode* t2) {
        if(!t1 || !t2) return t1? t1 : t2; // check if t1 or t2 is null. if t1 is null, !t1 = true
        
        TreeNode* node = new TreeNode(t1->val + t2-> val);
        node->left = mergeTrees(t1->left, t2->left);
        node->right = mergeTrees(t1->right, t2->right);
    }
};
```

// 方法2： 遍历
```
public:
    TreeNode* mergeTrees(TreeNode* t1, TreeNode* t2) {
        if(!t1) return t2;
        if(!t2) return t1;
        TreeNode* res = t1;
        stack<TreeNode*> s1, s2;
        s1.push(t1), s2.push(t2);
        while(!s1.empty()) {
            TreeNode* c1 = s1.top();
            TreeNode* c2 = s2.top();
            s1.pop(), s2.pop();
            c1->val += c2->val; 
            
            if(c1->left == NULL && c2->left != NULL) c1->left = c2->left;
            else if(c1->left != NULL && c2->left != NULL) s1.push(c1->left), s2.push(c2->left);
            
            if(c1->right == NULL && c2->right != NULL) c1->right = c2->right;
            else if(c1->right != NULL && c2->right != NULL) s1.push(c1->right), s2.push(c2->right);
        }
        return res;
    }
```

因为题目说不用create new node， 所以直接修改了t1，要叠加的node才放入stack，不用的直接pointer指过去就行
