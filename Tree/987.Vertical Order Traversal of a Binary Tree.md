class Solution {
    public List<List<Integer>> verticalTraversal(TreeNode root) {
        // pq->排序，注意 we report the values of the nodes in order from top to bottom (decreasing Y coordinates).
        TreeMap<Integer, TreeMap<Integer, PriorityQueue<Integer>>> map = new TreeMap<>();
        List<List<Integer>> res = new ArrayList<>();
        dfs(root, 0, 0, map);
        // 将 TreeMap 的东西放到res，因为TreeMap 的 key是排序的，像CPP的map
        for (TreeMap<Integer, PriorityQueue<Integer>> y : map.values()) {
            res.add(new ArrayList<>());
            for (PriorityQueue<Integer> nodes : y.values()) {
                while (!nodes.isEmpty())
                    res.get(res.size() - 1).add(nodes.poll());
            }
        }
        
        return res;
    }
    
    private void dfs(TreeNode root, int x, int y, TreeMap<Integer, TreeMap<Integer,
                     PriorityQueue<Integer>>> map) {
        if (root == null) return;
        // TreeMap<Integer, TreeMap<Integer, PriorityQueue<Integer>>>
        if (!map.containsKey(x))
            map.put(x, new TreeMap<>());
        if (!map.get(x).containsKey(y))
            map.get(x).put(y, new PriorityQueue<>());
        
        map.get(x).get(y).offer(root.val);
        dfs(root.left, x - 1, y + 1, map);
        dfs(root.right, x + 1, y + 1, map);
    }
}
