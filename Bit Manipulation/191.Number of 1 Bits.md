### Method 1: Brute force
```
class Solution {
public:
    int hammingWeight(uint32_t n) {
        int cnt = 0;
        while (n) {
            if (n % 2 == 1) cnt += 1;
            n /= 2;
        }
        return cnt;
    }
};
```

### Method 2: bit shifting
```
class Solution {
public:
    int hammingWeight(uint32_t n) {
        int cnt = 0;
        while (n) {
            cnt += (n & 1);
            n = n >> 1;
        }
        return cnt;
    }
};
```
