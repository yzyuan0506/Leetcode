```
class Solution {
public:
    string largestNumber(vector<int>& nums) {
        vector<string> temp;
        for (auto n : nums) // 将所有数字转换成string
            temp.push_back(to_string(n));
        
        sort(temp.begin(), temp.end(), 
             [](string &s1, string &s2) { return s1 + s2 > s2 + s1; });
        
        string res;
        for (auto s : temp)
            res += s;
        
        // remove leading 0s
        while (res[0] == '0' && res.length() > 1)
            res.erase(0, 1);
        
        return res;
    }
};
```

主要秀在 s1 + s2 > s2 + s1

说明 s1 > s2
