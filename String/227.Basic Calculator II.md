```
class Solution {
public:
    int calculate(string s) {
        stack<int> stk;
        
        char sign = '+';    // 很重要，预操作
        long num = 0;   // 要用long，不然会overflow

        for (int i = 0; i < s.length(); i++) { 
            if (isdigit(s[i]))
                num = num * 10 + s[i] - '0';
            if (!isdigit(s[i]) && !isspace(s[i]) || i == s.length() - 1) {
                if (sign == '+') {
                    stk.push(num);
                } else if (sign == '-') {
                    stk.push(-num);
                } else if (sign == '*') {
                    num = stk.top() * num;
                    stk.pop();
                    stk.push(num);
                } else if (sign == '/') {
                    num = stk.top() / num;
                    stk.pop();
                    stk.push(num);
                }
                sign = s[i];
                num = 0;
            }
        }
        
        int res = 0;
        while (!stk.empty()) {
            res += stk.top();
            stk.pop();
        }
        
        return res;
    }
};
```

注意这里用sign保存之前的运算符号，不是当前遇到的，当前遇到新的符号会留到后面用。。
