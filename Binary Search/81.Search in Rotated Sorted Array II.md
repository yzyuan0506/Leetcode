先看一下 33 题的解法：
题目说明没有duplicate

```
public int search(int[] nums, int target) {
    int start = 0;
    int end = nums.length - 1;
    while (start <= end) {
        int mid = (start + end) / 2;
        if (target == nums[mid]) {
            return mid;
        }
        //左半段是有序的
        if (nums[start] <= nums[mid]) {
            //target 在这段里
            if (target >= nums[start] && target < nums[mid]) {
                end = mid - 1;
            //target 在另一段里
            } else {
                start = mid + 1;
            }
        //右半段是有序的
        } else {
            if (target > nums[mid] && target <= nums[end]) {
                start = mid + 1;
            } else {
                end = mid - 1;
            }
        }

    }
    return -1;
}
```


### Method 1： BFS

根据上面的code，由于 nums[mid] 可能与 nums[start] 相等，
此时我们无法判断哪一段是有序的，于是我们就移动start 到下一位

```
class Solution {
public:
    bool search(vector<int>& nums, int target) {
        int start = 0, end = nums.size() - 1;
        
        while (start <= end) {
            
            int mid = (end + start) >> 1;
            
            if (nums[mid] == target || target == nums[0]) return true;
            
            if (nums[start] < nums[mid]) { // mid 的左半段是有序的
                if (target >= nums[start] && target < nums[mid]) {
                    end = mid - 1;
                } else {
                    start = mid + 1;
                }
                
            } else if (nums[start] == nums[mid]) {  // 因为有 duplicate 无法判断 mid 的哪边是有序的
                start++;
            } else {    // mid 的右半段是有序的
                if (target >= nums[mid] && target <= nums[end]) {
                    start = mid + 1;
                } else {
                    end = mid - 1;
                }
            }
        }
        
        return false;
    }
};
```

Avg Time Complexity:
O(log n)

Worst TIme COmplexity:
