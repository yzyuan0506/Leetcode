class Solution {
public:
    int findLengthOfLCIS(vector<int>& nums) {
        int n = nums.size();
        if (n == 0) return 0;
        
        int maxLen = INT_MIN;
        int len = 1;
        
        for (int r = 1; r < n; r++) {
            if (nums[r - 1] < nums[r]) {
                len += 1;
                maxLen = max(maxLen, len);
            } else {
                len = 1;
            }
        }
        
        maxLen = max(maxLen, len);
        
        return maxLen;
    }
};
