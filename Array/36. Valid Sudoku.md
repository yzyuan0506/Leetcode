### Method 1: Brute Force
```
class Solution {
public:
    bool isValidSudoku(vector<vector<char>>& board) {
        // 判断每一行
        for (int i = 0; i < 9; i ++) {
            if (!isValidRow(board[i]))
                return false;
        }
        // 判断每一列
        for (int i = 0; i < 9; i++) {
            if (!isValidCol(board, i))
                return false;
        }
        // 判断每个小九宫格, (i, j): 左上角第一个元素的坐标
        for (int i = 0; i < 9; i += 3)
            for (int j = 0; j < 9; j += 3) {
                if (!isValidBox(i, j, board))
                    return false;
            }
        
        return true;
    }
private:
    bool isValidRow(const vector<char> row) {
        int map[128] = {0};   // ASCII 只有 128 个
        for (char ch : row) {
            if (ch != '.') {
                if (map[ch] > 0) return false;
                map[ch] += 1;
            }
        }
        return true;
    }
    
    bool isValidCol(const vector<vector<char>>& board, int col) {
        int map[128] = {0};
        for (int i = 0; i < 9; i++) {
            char ch = board[i][col];
            if (ch != '.') {
                if (map[ch] > 0) return false;
                map[ch] += 1;
            }
        }
        return true;
    }
    
    bool isValidBox(int y, int x, const vector<vector<char>>& board) {
        int map[128] = {0};
        for (int i = 0; i < 3; i++)
            for (int j = 0; j < 3; j++) {
                char ch = board[y + i][x + j];
                if (ch != '.') {
                    if (map[ch] > 0) return false;
                    map[ch] += 1;
                }
            }
        return true;
    }
};
```
