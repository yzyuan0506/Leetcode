### Method 1
```
#include <iostream>

const int maxn = 10;
int nums[maxn];
bool bools[maxn] = {0};
int n;

using namespace std;

void dfs(int start) {
    if (start == n) {
        for (int i = 0; i < n; i++) cout << nums[i] << " ";
        cout << endl;
        return;
    }
    
    for (int i = 1; i <= n; i++) {
        if (bools[i - 1] == false) {
            // start 的 位置为开头，插入 1 到 n 的数字
            nums[start] = i;
            // 标记这个数已经被使用
            bools[i - 1] = true;
            
            dfs(start + 1);
            // 恢复
            bools[i - 1] = false;
        }
    }

}

int main() {
    cin >> n;
    
    dfs(0);
    
    return 0;
}
```

### Method 2: Backtracking from Jame Moen

虽然不能按照字典序输出所有组合，但是能输出全部组合
```
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

void backtracking(vector<int>& nums, int start) {
    if (start == nums.size()) {
        for (auto n : nums) cout << n << " ";
        cout << endl;
        return;
    } else {
        for (int i = start; i < nums.size(); i++) {
            swap(nums[start], nums[i]);
            backtracking(nums, start + 1);
            swap(nums[start], nums[i]);
        }
    }

}

int main() {
    int n;
    cin >> n;
    
    vector<int> nums(n, 0);
    for (int i = 1; i <= n; i++) nums[i - 1] = i;
    
    backtracking(nums, 0);
    
    return 0;
}
```
