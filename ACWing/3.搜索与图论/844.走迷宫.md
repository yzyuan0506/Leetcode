#include <iostream>
#include <queue>
#include <cstring>

using namespace std;

int n, m;
typedef pair<int, int> PII;
const int N = 110;

// matrix 是迷宫
// distances(ds) 记录到达每个点的最短距离,bfs第一次访问这个点时就是最短距离
int matrix[N][N], ds[N][N];

int bfs() {
    queue<PII> q;
    
    // -1 表示该点没有被访问过
    memset(ds, -1, sizeof ds);
    ds[0][0] = 0;
    matrix[0][0] = 0;
    q.push({0, 0});
    
    // dx, dy 表示上下左右移动
    int dx[4] = {-1, 0, 1, 0}, dy[4] = {0, 1, 0, -1};
    
    while (!q.empty()) {
        PII cur = q.front();
        q.pop();
        
        for (int i = 0; i < 4; i++) {
            int x = cur.first + dx[i];
            int y = cur.second + dy[i];
            
            if (x >= 0 && x < n && y >= 0 && y < m
                && matrix[x][y] == 0 && ds[x][y] == -1) {
                    // 当前能到的点最短距离等于上一步的 + 1
                    ds[x][y] = ds[cur.first][cur.second] + 1;
                    q.push({x, y});
                }
        }
    }
    
    return ds[n - 1][m - 1];
}

int main() {
    cin >> n >> m;
    for (int i = 0; i < n; i++)
        for (int j = 0; j < m;j++)
            cin >> matrix[i][j];
            
    cout << bfs() << endl;
    return 0;
}
