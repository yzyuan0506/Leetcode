```
#include <iostream>
#include <vector>

using namespace std;

int n, m;

int main() {
    cin >> n >> m;
    vector<int> dp(m + 1, 0);
    
    // 跟 01背包相比，多一层循环k，表示k 个第i件物品
    for (int i = 0; i < n; i++) {
        // v: 物品体积，w：物品价值，s：物品数量
        int v, w, s;
        cin >> v >> w >> s;
        for (int j = m; j >= 0; j--)
            for (int k = 1; k <= s; k++) {
                if (k * v <= j)
                    dp[j] = max(dp[j], dp[j - k * v] + k * w);
            }
    }
    
    cout << dp[m] << endl;
    
    return 0;
}
```
