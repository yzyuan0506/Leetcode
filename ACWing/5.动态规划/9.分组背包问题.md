```
#include <iostream>

using namespace std;

const int N = 110;

int n, m;
int dp[N], v[N], w[N];

int main() {
    cin >> n >> m;
    
    for (int i = 0; i < n; i++) {
        int s;  // 一个组有多少个物品
        cin >>s;
        
        for (int j = 0; j < s; j++) // 输入组内物品的体积和价值
            cin >> v[j] >> w[j];
            
        for (int j = m; j >= 0; j--)
            for (int k = 0; k < s; k++) { // 遍历分组内的所有物品
                if (v[k] <= j)
                    dp[j] = max(dp[j], dp[j - v[k]] + w[k]);
            }
    }
    
    cout << dp[m] << endl;
    
    return 0;
}
```
