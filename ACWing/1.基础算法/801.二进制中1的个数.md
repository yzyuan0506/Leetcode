```
#include<iostream>

using namespace std;

int main() {
    int n;
    cin >> n;
    
    while (n--) {
        int num, count = 0;
        cin >> num;
        
        // i & -i 等于 i 取反 再加1
        for (int i = num; i; i -= i & -i) count++;
        
        cout << count << " ";
    }
    
    return 0;
}
```
