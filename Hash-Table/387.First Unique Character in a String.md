```
class Solution {
public:
    int firstUniqChar(string s) {
        vector<int> counts(26, 0);  // 记录每个字母出现的次数
        for (char ch : s) {
            counts[ch - 'a']++;
        }
        for (int i = 0; i < s.size(); i++) {
            if (counts[s[i] - 'a'] == 1)
                return i;
        }
        
        return -1;
    }
};
```

用 vector 比用map 快得多！！！看到只有小写字母的string优先考虑用 vector
