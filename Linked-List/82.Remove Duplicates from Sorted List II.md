### Method 1: Hash Map
```
class Solution {
public:
    ListNode* deleteDuplicates(ListNode* head) {
        unordered_map<int,int> um;
        ListNode* curr = head;
        ListNode* dummy = new ListNode(-1);
        ListNode* prev = dummy;
        while (curr) {
            um[curr->val] += 1;
            curr = curr->next;
        }
        // reset
        curr = head;
        while (curr) {
            if (um[curr->val] == 1) {
                prev->next = curr;
                prev = curr;
            } else {
                prev->next = curr->next;
            }
            curr = curr->next;
        }
        return dummy->next;
    }
};
```

### Method 2: one iteration
发现原来链表是 sorted 的，所以可以不用hash map

```
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* deleteDuplicates(ListNode* head) {
        if(!head || !head->next) return head;
        ListNode* curr = head;
        ListNode* dummy = new ListNode(-1);
        ListNode* prev = dummy;
        bool flag = true;  // should the current head be added ?
        
        while (curr) {
            while (curr->next && curr->val == curr->next->val) {
                flag = false;
                curr = curr->next;
            }
            if (flag) {  // if should be added
                prev->next = curr;
                prev = curr;
            }
            curr = curr->next;
            flag = true;  // time for a new head value, set flag back to true
        }
        
        prev->next = nullptr;
        return dummy->next;
    }
};
```
