```
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        ListNode* curr = head;
        ListNode* next = NULL;
        ListNode* prev = NULL;  // 反转链表的头结点
        while(curr != NULL) {
            next = curr->next; // 记下原来的next node
            curr->next = prev;  // 当前的node->next 指向反转链表的头
            prev = curr;    // 更新反转链表的头
            curr = next;    // 到下一个遍历的node
        }
        return prev;
    }
};
```
用三个指针 prev，curr，next，时间复杂度O(n)，空间复杂度O(1)
