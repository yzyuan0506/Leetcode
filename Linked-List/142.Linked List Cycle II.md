```
public class Solution {
    public ListNode detectCycle(ListNode head) {
        if(head == null) {
            return null;
        }
        
        ListNode slow = head;
        ListNode fast = head;
        
        // detect cycle
        do {
            if(fast.next != null) {
                fast = fast.next.next;
                if(fast != null) {
                    slow = slow.next;
                } else {
                    // No cycle in the linked list
                    return null;
                }
            } else {
                return null;
            }
        } while(slow != fast);
        
        // find entry node
        ListNode node1 = head;
        ListNode node2 = slow;
        while (node1 != node2) {
            node1 = node1.next;
            node2 = node2.next;
        }
        
        return node1;
    }
}
```
