class Solution {
    public List<Integer> largestDivisibleSubset(int[] nums) {
        List<Integer> res = new ArrayList<>();
        int n = nums.length;
        if (n == 0) return res;
        
        Arrays.sort(nums);
        
        // dp[i]: considering nums[0:i],
        // the largest divisible subset ending with nums[i]
        int[] dp = new int[n];
        Arrays.fill(dp, 1);
        
        // 记录一个数字前面的（更小的），能跟它整除的数字，-1 表示没有
        int[] prev = new int[n];
        Arrays.fill(prev, -1);
        
        for (int i = 0; i < n; i++)
            for (int j = 0; j < i; j++) { // 找 ending with i 的最大subset
                if (nums[i] % nums[j] == 0) {
                    dp[i] = Math.max(dp[i], dp[j] + 1);
                    // 如果当前数字能有较大的subset，记录一下数字
                    if (dp[i] == dp[j] + 1)
                        prev[i] = j;
                }
            }
        
        int len = 0, maxIndex = 0;
        // 找出拥有最大subset 的 ending index
        for (int i = 0; i < n; i++) {
            if (dp[i] > len) {
                len = dp[i];
                maxIndex = i;
            }
        }
        
        // 将 maxIndex 前面的数加到最终的集合中
        for (int i = 0; i < len; i++) {
            res.add(nums[maxIndex]);
            maxIndex = prev[maxIndex];
        }
        
        return res;
    }
}
