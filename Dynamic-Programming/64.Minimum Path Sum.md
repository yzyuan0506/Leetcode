class Solution {
public:
    int minPathSum(vector<vector<int>>& grid) {
        int col = grid.size();
        int row = grid[0].size();
        if(col < 1 && row < 1) {
            return grid[0][0];
        }
        // initialize top row
        for(int i = 1; i < row; i++) {
            grid[0][i] += grid[0][i-1]; 
        }
        if(col == 1) return grid[0][row-1];
        
        // intialize left column
        for(int i = 1; i < col; i++) {
            grid[i][0] += grid[i-1][0]; 
        }
        if(row == 1) return grid[col-1][0];

        for(int i = 1; i < col; i++) {
            for(int j = 1; j < row; j++) {
                grid[i][j] += min(grid[i-1][j], grid[i][j-1]);
            }
        }
        return grid[col-1][row-1];
    }
};
