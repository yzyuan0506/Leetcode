class Solution {
public:
    bool isPerfectSquare(int n) {
        long int mid, l = 0, r = n, sq;
        while (l <= r) {
            mid = (l + r)/2;
            sq = mid * mid;
            if (sq  == n) {
                return true;
            } else if (sq > n) {
                r = mid - 1;
            } else {
                l = mid + 1;
            }
        }
        return false;
    }
};
