public:
void randomize(vector<int>& arr) {
    if (arr.empty()) return;
    int n = arr.size() - 1;
    vector<bool> bools(false, n + 1);

    for (int i = 0; i < n; i++) {
        if (bools[i]) continue;
        int random = rand() % n;
        while (random == i) random = rand() % n;
        swap(arr, i, random);
        bools[i] = true;
        bools[random] = true;
    }
}

private:
void swap(vector<int>& arr, int i, int j) {
    int temp = arr[i];
    arr[i] = arr[j];
    arr[j] = temp;
}
